// ---------------------------------------------------------------------------------------------------------------
// OpenGLFXManager.h  —  Visual Effects Manager for the OpenGL Renderer
// ---------------------------------------------------------------------------------------------------------------
// OpenGL equivalent of DX_FXManager.h.
// Replaces all D3D11 types with OpenGL equivalents:
//   ID3D11DeviceContext*  ->  (none; OpenGL is a state-machine — no command buffer)
//   ID3D11BlendState*     ->  GLuint shader program + glBlendFunc state
//   XMMATRIX              ->  per-platform matrix type (DirectXMath / GLM)
//
// Platform guards:
//   #if defined(_WIN32)  /  #elif defined(__linux__)  /  #elif defined(__ANDROID__)
// ---------------------------------------------------------------------------------------------------------------

enum class FXType {
    ColorFader,
    PixelOut,
    ScreenWipe,
    Scroller,
    ParticleExplosion,
    Starfield,
    TextScroller,
    WarpDotTunnel,
    TextFadeInOut,                                                              // Loading-screen text with per-frame fade in / fade out
};

enum class FXSubType {
    FadeIntoColor, FadeToBackground, FadeToTargetColor,
    WipeRight, WipeLeft, WipeUp, WipeDown,
    ScrollRight, ScrollLeft, ScrollUp, ScrollDown,
    ScrollUpAndLeft, ScrollUpAndRight, ScrollDownAndLeft, ScrollDownAndRight,
    TXT_SCROLL_LTOR, TXT_SCROLL_RTOL, TXT_SCROLL_CONSISTANT, TXT_SCROLL_MOVIE,
    TXT_FADE_IN,                                                                // Fade text in from startColor → endColor
    TXT_FADE_OUT,                                                               // Fade text out from endColor → fadeOutColor
};

struct GLCallbackEntry {
    int                    fxID = -1;
    std::function<void()>  callback;
    std::atomic<bool>      isExecuted{ false };
    std::chrono::steady_clock::time_point creationTime;

    GLCallbackEntry() { creationTime = std::chrono::steady_clock::now(); }
    GLCallbackEntry(int id, std::function<void()> cb)
        : fxID(id), callback(std::move(cb)) { creationTime = std::chrono::steady_clock::now(); }
    GLCallbackEntry(const GLCallbackEntry& o)
        : fxID(o.fxID), callback(o.callback),
          isExecuted(o.isExecuted.load()), creationTime(o.creationTime) {}
    GLCallbackEntry& operator=(const GLCallbackEntry& o) {
        if (this != &o) { fxID = o.fxID; callback = o.callback;
            isExecuted = o.isExecuted.load(); creationTime = o.creationTime; }
        return *this;
    }
    GLCallbackEntry(GLCallbackEntry&& o) noexcept
        : fxID(o.fxID), callback(std::move(o.callback)),
          isExecuted(o.isExecuted.load()), creationTime(o.creationTime)
    { o.fxID = -1; o.isExecuted = true; }
    GLCallbackEntry& operator=(GLCallbackEntry&& o) noexcept {
        if (this != &o) {
            fxID = o.fxID; callback = std::move(o.callback);
            isExecuted = o.isExecuted.load(); creationTime = o.creationTime;
            o.fxID = -1; o.isExecuted = true;
        }
        return *this;
    }
};

    void UpdateScrollSpeed(BlitObj2DIndexType textureIndex, int newSpeed);

    void ApplyScroller(GLFXItem& fxItem);

    void DestroyFadeShaderProgram();

    std::vector<int> CollectActiveTextScrollerIDs() const {
        std::vector<int> ids;
        ids.reserve(effects.size());
        for (const auto& fx : effects)
            if (fx.type == FXType::TextScroller)
                ids.push_back(fx.fxID);
        return ids;
    }

    bool HasActiveFadeEffects() const {
        for (const auto& fx : effects)
            if (fx.type == FXType::ColorFader && fx.progress < 1.0f)
                return true;
        return false;
    }

    void SafelyClearAllEffects() {
        std::vector<GLFXItem>        tmp;  tmp.swap(effects);
        std::vector<GLCallbackEntry> tmpc; tmpc.swap(m_pendingCallbacks);
    }

    // OpenGL resources for fullscreen fade quad
    GLuint m_fadeShaderProgram = 0;
    GLuint m_fadeVAO           = 0;
    GLint  m_fadeColorUniform  = -1;

    // Atomic re-entrancy guard
    std::atomic<bool>            m_isRendering{ false };
    std::mutex                   m_effectsMutex;
    std::vector<GLCallbackEntry> m_pendingCallbacks;
    std::vector<GLScrollTween>   m_activeTweens;
    GLActiveFXState              m_savedFXState;

    // Full-effect snapshot for scene transitions
    std::vector<GLFXItem> m_sceneSavedEffects;
    int                   m_sceneSavedStarfieldID = 0;
    int                   m_sceneSavedTunnelID    = 0;
};
