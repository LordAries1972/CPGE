// ---------------------------------------------------------------------------------------------------------------
// Per-frame Vulkan resources
// ---------------------------------------------------------------------------------------------------------------
struct VulkanFrameData
{
    VkCommandBuffer               commandBuffer  = VK_NULL_HANDLE;
    VkSemaphore                   imageAvailable = VK_NULL_HANDLE;
    // renderFinished is per-swapchain-image (m_renderFinishedSemaphores), NOT per frame-in-flight.
    // Indexing by frame caused VUID-vkQueueSubmit-pSignalSemaphores-00067 when the swapchain
    // had more images than frames-in-flight and a semaphore was reused before the presentation
    // that waited on it had finished.
    VkFence                       inFlightFence  = VK_NULL_HANDLE;
    std::vector<VkDescriptorSet>  pendingFreeSets; // freed after fence signals next frame
};

// ---------------------------------------------------------------------------------------------------------------
// DeltaTimeSmoothing  (identical to DX11Renderer version — platform-agnostic)
// ---------------------------------------------------------------------------------------------------------------
class VKDeltaTimeSmoothing
{
    static constexpr int   HISTORY_SIZE     = 8;
    static constexpr float MAX_DELTA_VARIANCE = 0.004f;
    float deltaHistory[HISTORY_SIZE];
    int   historyIndex    = 0;
    float smoothedDelta   = 0.0f;
    float frameTargetTime = 1.0f / 60.0f;
    float accumulatedError = 0.0f;
    bool  isInitialized   = false;

public:
    VKDeltaTimeSmoothing()
    {
        for (int i = 0; i < HISTORY_SIZE; ++i) deltaHistory[i] = frameTargetTime;
    }

    float ProcessDelta(float rawDelta, float refreshRate)
    {
        frameTargetTime = 1.0f / refreshRate;
        float clamped   = std::clamp(rawDelta, frameTargetTime * 0.5f, frameTargetTime * 2.0f);
        deltaHistory[historyIndex] = clamped;
        historyIndex = (historyIndex + 1) % HISTORY_SIZE;
        if (!isInitialized)
        {
            if (historyIndex >= HISTORY_SIZE - 1) isInitialized = true;
            smoothedDelta = clamped;
            return smoothedDelta;
        }
        float weightedSum = 0.0f, totalWeight = 0.0f;
        for (int i = 0; i < HISTORY_SIZE; ++i)
        {
            float w = static_cast<float>(i + 1) / HISTORY_SIZE;
            int   idx = (historyIndex - 1 - i + HISTORY_SIZE) % HISTORY_SIZE;
            weightedSum += deltaHistory[idx] * w;
            totalWeight += w;
        }
        float avg = weightedSum / totalWeight;
        accumulatedError += (frameTargetTime - avg) * 0.1f;
        smoothedDelta = avg + std::clamp(accumulatedError, -MAX_DELTA_VARIANCE, MAX_DELTA_VARIANCE);
        return smoothedDelta;
    }

    float GetSmoothedDelta() const { return smoothedDelta; }

    void Reset()
    {
        historyIndex = 0; accumulatedError = 0.0f; isInitialized = false;
        for (int i = 0; i < HISTORY_SIZE; ++i) deltaHistory[i] = frameTargetTime;
    }
};

    VulkanTexture CreateTextureFromRGBA(const uint8_t* pixels, uint32_t width, uint32_t height,
                                        VkFormat format = VK_FORMAT_R8G8B8A8_SRGB);

    void CreateImage(uint32_t width, uint32_t height, VkFormat format, VkImageTiling tiling,
                     VkImageUsageFlags usage, VkMemoryPropertyFlags props,

    VkShaderModule  CreateShaderModuleFromSPIRV(const std::vector<uint32_t>& spirv) const;
    VkShaderModule  CreateShaderModuleFromGLSL(const std::string& glsl, VkShaderStageFlagBits stage) const;

    // Scene rendering
    inline void RenderGamePlay(float deltaTime);

    // Validation layers
    bool          CheckValidationLayerSupport() const;

        VkDebugUtilsMessageSeverityFlagBitsEXT severity,

        void* pUserData);
