      <!-- Define your Master and SubRelease versions -->

      <SubReleaseVersion>0</SubReleaseVersion>
      <VersionFilePath>$(ProjectDir)Version.id</VersionFilePath>
      <DefaultBuildNumber>695</DefaultBuildNumber>
    </PropertyGroup>
    <!-- Read existing version file if it exists -->
    <ReadLinesFromFile File="$(VersionFilePath)" Condition="Exists('$(VersionFilePath)')">
      <Output TaskParameter="Lines" ItemName="ExistingVersionLines" />
    </ReadLinesFromFile>
    <!-- Extract current build number from existing file -->
    <PropertyGroup>
      <ExistingVersionLine>@(ExistingVersionLines)</ExistingVersionLine>
      <!-- Extract number after 'v' and last dot -->
      <CurrentBuildStr>$([System.Text.RegularExpressions.Regex]::Match('$(ExistingVersionLine)', 'v\d+\.\d+\.(\d+)').Groups[1].Value)</CurrentBuildStr>
      <CurrentBuild Condition="'$(CurrentBuildStr)' != ''">$(CurrentBuildStr)</CurrentBuild>
      <CurrentBuild Condition="'$(CurrentBuildStr)' == ''">$(DefaultBuildNumber)</CurrentBuild>
      <!-- Increment build number -->
      <NewBuildNumber>$([MSBuild]::Add($(CurrentBuild), 1))</NewBuildNumber>
      <!-- Create new version string -->
      <NewVersionString>v$(MasterVersion).$(SubReleaseVersion).$(NewBuildNumber)</NewVersionString>
      <VersionFileContent>Current Build Version: $(NewVersionString)</VersionFileContent>
    </PropertyGroup>
    <!-- Write updated version to file -->
    <WriteLinesToFile File="$(VersionFilePath)" Lines="$(VersionFileContent)" Overwrite="true" Encoding="UTF-8" />
    <!-- Update BuildInfo.h with new version numbers (pure MSBuild, no external process) -->
    <ItemGroup>
      <_BuildInfoLines Include="#pragma once" />
      <_BuildInfoLines Include="// Authoritative build identity - update these on every release." />
      <_BuildInfoLines Include="constexpr int CURRENT_BUILD_VERSION    = $(MasterVersion)%3B" />
      <_BuildInfoLines Include="constexpr int CURRENT_BUILD_SUBVERSION = $(SubReleaseVersion)%3B" />
      <_BuildInfoLines Include="constexpr int CURRENT_BUILD            = $(NewBuildNumber)%3B" />
    </ItemGroup>
    <WriteLinesToFile File="$(ProjectDir)BuildInfo.h" Lines="@(_BuildInfoLines)" Overwrite="true" Encoding="UTF-8" />
