  - using XMFLOAT2 = Vector2;
  - using XMFLOAT3 = Vector3;
  - using XMFLOAT4 = Vector4;
  - using XMMATRIX = Matrix4x4;
  - using XMVECTOR = Vector4;
  - #define XM_PI (if not already defined)
  - namespace DirectX {} empty namespace so 'using namespace DirectX' compiles cleanly
- Guard: #if !defined(__USE_DIRECTX_11__) && !defined(__USE_DIRECTX_12__)
- Purpose: allows all code using XMFLOAT2/3/4/XMMATRIX/XMVECTOR to compile on OpenGL
